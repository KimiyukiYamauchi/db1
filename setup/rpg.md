「rpg」データベースに「パーティー」、「イベント」、「経験イベント」、「コード」テーブルを作成する

```sql
-- 既にデータベースがある場合はいったん削除、ない場合はエラー
drop database if exists `rpg`;

-- データベース作成（文字コードはUTF8MB4で日本語対応）
CREATE DATABASE `rpg` CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci;

-- 使用するデータベースを指定
USE `rpg`;

-- パーティーテーブル作成
CREATE TABLE パーティー (
  ID CHAR(3) PRIMARY KEY,             -- 英字1桁 + 数字2桁
  名称 VARCHAR(20) NOT NULL,
  職業コード CHAR(2) NOT NULL,       -- 01:勇者, 10:戦士, 11:武道家, 20:魔法使い, 21:学者
  HP INTEGER NOT NULL,
  MP INTEGER NOT NULL,
  状態コード CHAR(2) NOT NULL        -- 00:正常, 01:眠り, 02:毒, 03:沈黙, 04:混乱, 09:気絶
);

--- パーティーテーブルのデータの確認
select * from パーティー;

-- イベントテーブル作成
CREATE TABLE イベント (
  イベント番号 INTEGER PRIMARY KEY,            -- 主キー
  イベント名称 VARCHAR(50) NOT NULL,           -- イベント名（必須）
  タイプ CHAR(2) NOT NULL,                     -- 01: 強制, 02: フリー, 03: 特殊
  前提イベント番号 INTEGER,                   -- 事前にクリアが必要なイベント番号（外部キーになる可能性あり）
  後続イベント番号 INTEGER                    -- 次に発生するイベント番号（外部キーになる可能性あり）
);

-- 経験イベントテーブル作成
CREATE TABLE 経験イベント (
  イベント番号 INTEGER PRIMARY KEY,         -- イベントごとの識別番号
  クリア区分 CHAR(1) NOT NULL,              -- 0: プレイ中 / 1: クリア済
  クリア結果 CHAR(1),                      -- ランク（A, B, C）※未クリアはNULL
  ルート番号 INTEGER                       -- クリアしたイベントの連番 ※未クリアはNULL
);

-- コードテーブル作成
CREATE TABLE コード (
  コード種別 INTEGER,                -- 1: 職業コード, 2: 状態コード, 3: イベントタイプ, 4: クリア結果
  コード値 CHAR(2),                 -- 種別ごとのコード値（例: '01', '02' など）
  コード名称 VARCHAR(100),          -- コードの日本語名
  PRIMARY KEY (コード種別, コード値)
```
